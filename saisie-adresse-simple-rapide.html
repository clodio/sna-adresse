Jquery

code d'intégration jquery

Exemple d'intégration HTML+Javascript 
<!DOCTYPE html>
<html>
  <head>
    <title>La Poste : Service d'autocompletion d'adressen Serca jquery</title>
    <meta name="viewport" content="initial-scale=1.0, user-scalable=no">
    <meta charset="utf-8">
  </head>

  <body>
    <div id="addressSearch">
      <input id="addressSearchAutocomplete" placeholder="Entrez votre adresse"
             onFocus="geolocate()" type="text"></input>
    </div>
    <script>
      // Cet exemple montre l'intégration d'un formulaire de saisie d'adresse avec serca en utilisant jquery, pour d'autres solutions, voir http://www.laposte.fr/sna/adresses.
      $(function(){
         var options= {
            row3: true, // traitement en prenant en compte les lignes 3 d'adresse
            geolocation: true, // renvoie les coordonnées géolocalisées
            cedex: false, // prise en compte des adresses cedex
         };
         $("#addressSearchAutocomplete").geocomplete(options);
      }
    </script>

    <script src="https://ajax.googleapis.com/ajax/libs/jquery/3.1.1/jquery.min.js"></script>

    <script src="https://api.laposte.fr/address/v1/ressources/jquery.autocomplete.js?key=YOUR_API_KEY</script>
  </body>
</html>
